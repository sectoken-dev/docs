    创建Btc多签钱包

    通过平台快速创建一个2-3多签Btc钱包

    import requests

requests.post(
    url="https://testnet.sectoken.io/api/v1/wallet/btc/create/", 
    headers={"HMAC": "your token"}, 
    json={
        "pubkeys": [
            "020fcbb44ff4a1f423499b1eb9b2bd9cbe467934467d4857451c7eab7f8cb2885b", 
            "038f2d1612ae08a9595fbe1d30c9b13e748a4ae55c06d5d3fa5c0b2e59712d3b6b"
        ],
        "privkeys":[
            "your encrypted privkey1", 
        "your encrypted privkey2", 
        ],
        "wallet_name": "BtcTestWallet",
    }
)

    curl "https://testnet.sectoken.io/api/v1/wallet/btc/create/" \
    -H "HMAC:your token" \
    -H "Content-Type:application/json" \
    -X POST \
    -d '{
    "pubkeys": [
        "020fcbb44ff4a1f423499b1eb9b2bd9cbe467934467d4857451c7eab7f8cb2885b", 
        "038f2d1612ae08a9595fbe1d30c9b13e748a4ae55c06d5d3fa5c0b2e59712d3b6b"
    ],
    "privkeys": [
        "your encrypted privkey1",
        "your encrypted privkey2"
    ], 
    "wallet_name": "BtcTestWallet",
}'

    var request = require('request');

request({
    url: 'https://testnet.sectoken.io/api/v1/wallet/btc/create/',
    method: "post",
    headers: {
        'Content-Type': 'application/json',
        'HMAC': 'your token',
    },
    body: JSON.stringify({
        'pubkeys': [
            '02f9ecb53c10e999fd2ca3a8f4503d2c5df3521e38cef44570e34f7bdc23219c70',
            '023a602328d823875cd8a00393dbade47bd1f16dab100ad89ee35b1bc61607f10c'
        ],
        "privkeys": [
            "your encrypted privkey1",
            "your encrypted privkey2"
        ], 
        "wallet_name": "BtcTestWallet",
    })
});

    HTTP Request

    POST https://testnet.sectoken.io/api/v1/wallet/btc/create/

    请求参数

    参数名 参数类型 必填 说明
    pubkeys array 是 公钥列表
    privkeys array 否 私钥加密密文列表, 每个被上传的密文都必须经过base64编码
    wallet_name string 是 钱包名

    返回参数

    参数名 参数类型 说明
    code integer 返回码，200成功，其他失败，请查看错误码

    参数名 参数类型 说明
    wallet_id string 钱包的id
    activate_code string 钱包激活码
    server_pubkey string 钱包的xpub

    创建Btc多用户钱包

    通过该接口可以快速创建Btc钱包, 并将钱包更改为任意多签形式的多用户钱包 其他用户可以通过此接口返回的 invitation_code 加入钱包

    import requests

requests.post(
    url="https://testnet.sectoken.io/api/v1/wallet/btc/share_create/", 
    headers={"HMAC": "your token"}, 
    json={
            "pubkey":"02d654b7b6829df777a32267d366b4c24fd30fc1a61d816e043104edd198678e52", 
            "privkey": "your encrypted privkey",
            "threshold":2,
            "limit":2,
            "control": 1,
            "wallet_name": "BtcTestWallet",
    }
)

    curl "https://testnet.sectoken.io/api/v1/wallet/btc/share_create/"
    -H "HMAC:your token"
    -H "Content-Type:application/json"
    -X POST
    -d '{   
        "pubkey":"02d654b7b6829df777a32267d366b4c24fd30fc1a61d816e043104edd198678e52", 
        "privkey": "your encrypted privkey",
        "threshold":2,
        "limit":2,
        "control": 1,
        "wallet_name": "BtcTestWallet",
    }'

    var request = require('request');

request({
    url: 'https://testnet.sectoken.io/api/v1/wallet/btc/share_create/',
    mbtcod: "post",
    headers: {
        'Content-Type': 'application/json',
        'HMAC': 'your token',
    },
    body: JSON.stringify({
        "pubkey":"02d654b7b6829df777a32267d366b4c24fd30fc1a61d816e043104edd198678e52", 
        "privkey": "your encrypted privkey",
        "threshold":2,
        "limit":2,
        "control": 1,
        "wallet_name": "BtcTestWallet",
    })
});

    返回示例

    {
    "code": 200,
    "data": {
        "invitation_code": "cf761ad1b6c5f30a1557d942cb557b07"
    }
}

    HTTP Request

    POST https://testnet.sectoken.io/api/v1/wallet/btc/share_create/

    请求参数

    参数名 参数类型 必填 说明
    pubkey array 是 公钥
    privkey array 否 私钥, 必须BASE64格式
    wallet_name string 是 钱包名
    threshold integer 是 钱包最少签名个数,最大为7
    limit integer 是 钱包公钥总个数,最大为7
    control integer 是 0:平台参与签名; 1:平台不参与签名
    multi integer 否 是否允许单用户持有多个公钥，1为允许，默认值为0

    返回参数

    参数名 参数类型 说明
    code integer 返回码，200成功，其他失败，请查看错误码
    data object 返回内容

    data 参数

    参数名 参数类型 说明
    invitation_code string 分享码，其他用户输入此码可加入此多用户钱包

    加入多用户钱包

    该接口通过 invitation_code 可以加入他人创建好的钱包 

    import requests

requests.post(
    url="https://testnet.sectoken.io/api/v1/wallet/btc/join_wallet/",
    headers={"HMAC": "your token"},
    json={
        "invitation_code": "cf761ad1b6c5f30a1557d942cb557b07",
        "pubkey": "02d654b7b6829df777a32267d366b4c24fd30fc1a61d816e043104edd198678e52", 
        "privkey": "your encrypted privkey",
    } 
)

    curl "https://testnet.sectoken.io/api/v1/wallet/btc/join_wallet/" \
    -H "HMAC:your token" \
    -H "Content-Type:application/json" \
    -X POST \
    -d '{
        "invitation_code": "cf761ad1b6c5f30a1557d942cb557b07",
        "pubkey": "02d654b7b6829df777a32267d366b4c24fd30fc1a61d816e043104edd198678e52", 
        "privkey": "your encrypted privkey",
    }'

    var request = require('request');

request({
        url: 'https://testnet.sectoken.io/api/v1/wallet/btc/join_wallet/',
        mbtcod: "post",
        headers: {
            'Content-Type': 'application/json',
            'HMAC': 'your token',
        },
        body: JSON.stringify({
            "invitation_code": "cf761ad1b6c5f30a1557d942cb557b07",
            "pubkey": "02d654b7b6829df777a32267d366b4c24fd30fc1a61d816e043104edd198678e52", 
            "privkey": "your encrypted privkey",
        })
    });

    激活钱包

    通过创建钱包时返回的 wallet_id,activate_code 参数来激活并使用钱包

    import requests

requests.post(
    url="https://testnet.sectoken.io/api/v1/wallet/btc/activate/",
    headers={"HMAC": "your token"},
    json={
        "wallet_id": "cf761ad1b6c5f30a1557d942cb557b07",
        "activate_code": "712968",
    } 
)

    curl "https://testnet.sectoken.io/api/v1/wallet/btc/activate/" \
    -H "HMAC:your token" \
    -H "Content-Type:application/json" \
    -X POST \
    -d '{
        "wallet_id": "cf761ad1b6c5f30a1557d942cb557b07", 
        "activate_code": "712968"
    }'

    var request = require('request');

request({
        url: 'https://testnet.sectoken.io/api/v1/wallet/btc/activate/',
        method: "post",
        headers: {
            'Content-Type': 'application/json',
            'HMAC': 'your token',
        },
        body: JSON.stringify({
            'wallet_id': 'cf761ad1b6c5f30a1557d942cb557b07',
            'activate_code': '712968'

        })
    }
);

    交易Callback通知

    当用户钱包发生交易时，会向指定的url发送HTTP POST请求，请求头content-type为application/json。

    此url由用户指定，用户须部署服务接收HTTP POST请求。同时需先添加钱包Webhook。

    请求头中会有Checksum参数，用于验证请求信息是否完整， 通过 sha256（secret_key+request.body）来进行哈希.

    secret_key 为创建webhook时提交的secret_key, 如果创建时没有上传,默认为空字符串.

    请求headers示例 json { "Content_type": "application/json", "Checksum": "sha256=6ca13d52ca70c883e0f0bb101e425a89e8624de51db2d2392593af6a84118090" }

    output array 目的地址及金额或收款地址及金额, 具体看交易类型
    type string 交易类型,收款或付款

    decimal integer 币种精度
    memo string 区块链上可查的交易单备注信息
    webhook_id string callback 的 id

    错误码

    当发生错误时返回的错误码

    错误码 错误说明
    4000 请求参数错误
    4004 未找到请求资源
    5000 未知错误, 请联系Sectoken
    4201 钱包不存在
    4202 钱包冻结中
    4203 钱包下没有此地址
    4204 钱包余额不足
    4209 钱包激活码错误
    4210 钱包Webhook数目超限
    4211 地址格式错误
    4401 非法操作
    4402 此钱包不存在
    4403 钱包个数达到上限
    4404 用户权限不足
    4405 用户不存在
    4406 用户已在钱包中
    4407 用户不在钱包中
    4503 不能转账给钱包自己
    4502 禁止转向当前地址
    4503 目标地址不能为自己
    4504 签名错误
    4505 全部转出只能有一个收款地址

    4507 私钥密文必须是base64格式
    4508 webhook 不存在
